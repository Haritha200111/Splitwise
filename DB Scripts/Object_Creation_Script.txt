---Table Scripts

CREATE TABLE IF NOT EXISTS public.useraccount
(
    firstname character varying(255) COLLATE pg_catalog."default",
    lastname character varying(255) COLLATE pg_catalog."default",
    emailid character varying(255) COLLATE pg_catalog."default",
    userpassword character varying(255) COLLATE pg_catalog."default",
    CONSTRAINT emailid_ukey UNIQUE (emailid)
)
-----------------------------------------

CREATE TABLE IF NOT EXISTS public.userexpense
(
    emailid text COLLATE pg_catalog."default",
    amount integer,
    group_id integer,
    issettled boolean DEFAULT false,
    group_name text COLLATE pg_catalog."default"
)

------------------------------------------

CREATE TABLE IF NOT EXISTS public.usergroup
(
    group_id integer NOT NULL GENERATED BY DEFAULT AS IDENTITY ( INCREMENT 1 START 1 MINVALUE 1 MAXVALUE 2147483647 CACHE 1 ),
    group_name character varying(255) COLLATE pg_catalog."default",
    group_expense integer,
    created_by text COLLATE pg_catalog."default",
    users_to_add text[] COLLATE pg_catalog."default",
    CONSTRAINT usergroup_pkey PRIMARY KEY (group_id),
    CONSTRAINT groupname_ukey UNIQUE (group_name)
)
